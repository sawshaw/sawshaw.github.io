```python
import urllib.request
import urllib.parse
import os

# 保存到的文件夹
dir_path = "D:/pythonFile/"

# ts视频流的基础路径
base_url = "https://vip.aqdw40.com/videos/tag/zipai"
playlist_url = "https://cdn.aqd-tv.com/video/aae7b6bd/playlist/"
play_url = "https://vip.aqdw40.com/videos/play/26314"

# 文件名的通用部分
base_name = "b3ba5a%s.ts"

# 分析ts视频流文件名称规律,  比如后缀是按数字排序的, 依次爬取(具体问题具体分析)
for i in range(100, 140):
    name_num = '%06d' % i
    name = base_name % name_num
    file_path = os.path.join(dir_path, name)
    print(name)
    urllib.request.urlretrieve(playlist_url + name, file_path)
```

合并：
```python
import os

base_path = "D:/pythonFile"
save_path = 'D:/pythonFile'

# 获取ts文件目录
file_list = os.listdir(base_path)

# 按后缀数字排序ts文件
#这段代码是对一个文件列表进行排序，排序的依据是文件名中的第12个字符到倒数第4个字符所表示的数字。
#其中lambda函数是一个匿名函数，用于对每个文件名进行处理并返回排序依据。
file_list.sort(key=lambda x: int(x[11:len(x) - 3]))

# 以二进制打开目标文件
with open(save_path + '/pythonFile.ts', "ab+") as target:
    # 依次将数据写入目标文件
    for temp in file_list:
        with open(base_path + '/' + temp, "rb+") as temp_file:
            target.write(temp_file.read())
        print(temp)
```
